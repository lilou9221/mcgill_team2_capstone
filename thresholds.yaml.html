# Biochar Suitability Thresholds
# Thresholds for soil properties used in suitability scoring

# Soil Moisture (Volume fraction, 0-0.9)
soil_moisture:
  # Optimal range: 0.1-0.3
  # Below 0.1: dry (low score)
  # Above 0.3: saturated (low score)
  optimal_min: 0.1
  optimal_max: 0.3
  acceptable_min: 0.0
  acceptable_max: 0.9
  scoring:
    high: [0.1, 0.3]  # Score 8-10: Optimal range
    medium: [0.0, 0.1]  # Score 5-7: Below optimal (dry)
    low: [0.3, 0.9]  # Score 0-4: Above optimal (saturated)

# Soil Temperature (Kelvin)
soil_temperature:
  # Healthy range: 5-35°C for most plant roots
  # <5°C inhibits root growth
  # <0°C no biotic activity
  # Converted to Kelvin: 5°C = 278.15K, 35°C = 308.15K, 0°C = 273.15K
  optimal_min: 278.15  # 5°C in Kelvin
  optimal_max: 308.15  # 35°C in Kelvin
  acceptable_min: 273.15  # 0°C in Kelvin (minimum for biotic activity)
  acceptable_max: 308.15
  scoring:
    high: [278.15, 308.15]  # Score 8-10: 5-35°C (optimal for plant roots)
    medium: [273.15, 278.15]  # Score 5-7: 0-5°C (inhibits root growth)
    low: [0.0, 273.15]  # Score 0-4: <0°C (no biotic activity)

# Soil Organic Carbon (g/kg)
soil_organic_carbon:
  # Healthy range: 3-9% (30-90 g/kg) for dark grey, >12% (120 g/kg) for black
  # Data range: 0-120 g/kg
  optimal_min: 30.0  # 3% = 30 g/kg
  optimal_max: 120.0  # 12% = 120 g/kg
  acceptable_min: 0.0
  acceptable_max: 120.0
  scoring:
    high: [90.0, 120.0]  # Score 8-10: >9% (>90 g/kg) - high organic matter
    medium: [30.0, 90.0]  # Score 5-7: 3-9% (30-90 g/kg) - moderate organic matter
    low: [0.0, 30.0]  # Score 0-4: <3% (<30 g/kg) - low organic matter

# Soil pH
soil_pH:
  # Healthy range: 5.5-7.5
  # <4.2 indicates Dystric (acidic, nutrient-poor) conditions
  optimal_min: 5.5
  optimal_max: 7.5
  acceptable_min: 4.2
  acceptable_max: 7.5
  scoring:
    high: [5.5, 7.5]  # Score 8-10: Optimal pH range
    medium: [4.2, 5.5]  # Score 5-7: Acceptable but suboptimal
    low: [0.0, 4.2]  # Score 0-4: <4.2 (Dystric/acidic, nutrient-poor)

# Note: Land cover and soil type are available in data but not used in initial scoring
# They will be handled in later steps
